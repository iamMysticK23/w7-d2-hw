Question 1
Write function bmi that calculates body mass index
(bmi = weight / height2)
if bmi <= 18.5 return "Underweight"
<= 25.0 return "Normal"
if bmi <= 30.0 return "Overweight"
bmi > 30 return "Obese"



Solution:
function bmi(weight, height) {

const bmi = weight / (height * height);

// BMI Categories

if (bmi <= 18.5) {
return "Underweight";
} else if (bmi <= 25.0) {
return "Normal";
} else if (bmi <= 30.0) {
return "Overweight";
} else {
return "Obese";

}
}
